Create a Grimoire (Class)

grim MagicalCrow:
    init(name, power):
        self.name = name
        self.power = power
        self.spells = []
    
    spell learn(spell_name):
        self.spells.append(spell_name)
        print(f"{self.name} learned {spell_name}!")
    
    spell cast(target):
        if len(self.spells) > 0:
            spell = self.spells[0]
            return f"{self.name} casts {spell} at {target}!"

// Create and use a magical crow
munin = MagicalCrow("Munin", 100)
munin.learn("Lightning Bolt")
print(munin.cast("darkness"))

Error Handling with Magic

// Carrion's unique error handling
attempt:
    risky_spell = cast_dangerous_magic()
    print("Spell succeeded!")
ensnare MagicError as err:
    print(f"Magic failed: {err}")
resolve:
    print("Cleaning up magical residue...")

// Handle multiple error types
attempt:
    result = divide_by_zero()
ensnare DivisionError:
    print("Cannot divide by void!")
ensnare ValueError:
    print("Invalid magical input!")

Install from Source

# Clone the repository
git clone https://github.com/javanhut/TheCarrionLanguage
cd TheCarrionLanguage

# Build from source
cd src && go build -o carrion

# Run Carrion
./carrion
