Fibonacci Sequence Generator

Generate the mystical Fibonacci sequence.

spell fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

// Generate first 10 Fibonacci numbers
print("The Fibonacci Sequence:")
for i in range(10):
    print(f"F({i}) = {fibonacci(i)}")

Magic 8-Ball

Ask the mystical 8-ball for guidance!

import "random"

grim Magic8Ball:
    init():
        self.responses = [
            "It is certain",
            "Without a doubt",
            "Yes, definitely",
            "Reply hazy, try again",
            "Ask again later",
            "Cannot predict now",
            "Don't count on it",
            "My sources say no",
            "Outlook not so good",
            "Very doubtful"
        ]

    spell shake():
        index = random.randint(0, len(self.responses) - 1)
        return self.responses[index]

ball = Magic8Ball()
question = input("Ask the Magic 8-Ball a question: ")
print(f"The 8-Ball says: {ball.shake()}")

Simple Calculator Grimoire

A calculator with magical operations.

grim Calculator:
    init():
        self.result = 0

    spell add(x, y):
        self.result = x + y
        return self.result

    spell subtract(x, y):
        self.result = x - y
        return self.result

    spell multiply(x, y):
        self.result = x * y
        return self.result

    spell divide(x, y):
        if y == 0:
            return "Cannot divide by zero!"
        self.result = x / y
        return self.result

    spell power(base, exp):
        self.result = base ** exp
        return self.result

calc = Calculator()
print(f"10 + 5 = {calc.add(10, 5)}")
print(f"10 - 5 = {calc.subtract(10, 5)}")
print(f"10 * 5 = {calc.multiply(10, 5)}")
print(f"10 / 5 = {calc.divide(10, 5)}")
print(f"2 ^ 10 = {calc.power(2, 10)}")
Output:
10 + 5 = 15
10 - 5 = 5
10 * 5 = 50
10 / 5 = 2.0
2 ^ 10 = 1024

Text Adventure Game

A simple text-based adventure in Carrion.

grim Room:
    init(name, description):
        self.name = name
        self.description = description
        self.exits = {}
        self.items = []

    spell add_exit(direction, room):
        self.exits[direction] = room

    spell add_item(item):
        self.items = append(self.items, item)

    spell describe():
        print(f"\n=== {self.name} ===")
        print(self.description)
        if len(self.items) > 0:
            print(f"Items here: {self.items}")
        print(f"Exits: {keys(self.exits)}")

// Create rooms
hall = Room("Great Hall", "A grand hall with torches on the walls.")
library = Room("Library", "Dusty books line the shelves.")
garden = Room("Garden", "A peaceful garden with magical herbs.")

// Connect rooms
hall.add_exit("north", library)
hall.add_exit("east", garden)
library.add_exit("south", hall)
garden.add_exit("west", hall)

// Add items
library.add_item("Ancient Spellbook")
garden.add_item("Healing Herb")

// Game loop
current_room = hall
print("Welcome to the Mystical Manor!")

while True:
    current_room.describe()
    command = input("\nWhat do you do? ").lower()

    if command == "quit":
        print("Thanks for playing!")
        break
    elif command in current_room.exits:
        current_room = current_room.exits[command]
    else:
        print("You can't go that way!")

Number Guessing Game

Can you guess the mystical number?

import "random"

spell play_game():
    secret = random.randint(1, 100)
    attempts = 0
    max_attempts = 7

    print("I'm thinking of a number between 1 and 100.")
    print(f"You have {max_attempts} attempts to guess it!")

    while attempts < max_attempts:
        guess = int(input("Your guess: "))
        attempts = attempts + 1

        if guess == secret:
            print(f"Congratulations! You got it in {attempts} attempts!")
            return True
        elif guess < secret:
            print("Too low! Try higher.")
        else:
            print("Too high! Try lower.")

        remaining = max_attempts - attempts
        if remaining > 0:
            print(f"Attempts remaining: {remaining}")

    print(f"Game over! The number was {secret}")
    return False

play_game()

ASCII Art Generator

Create simple ASCII art patterns.

spell draw_triangle(height):
    for i in range(1, height + 1):
        spaces = " " * (height - i)
        stars = "*" * (2 * i - 1)
        print(spaces + stars)

spell draw_rectangle(width, height):
    for row in range(height):
        if row == 0 or row == height - 1:
            print("*" * width)
        else:
            print("*" + " " * (width - 2) + "*")

spell draw_diamond(size):
    // Top half
    for i in range(1, size + 1):
        spaces = " " * (size - i)
        stars = "*" * (2 * i - 1)
        print(spaces + stars)
    // Bottom half
    for i in range(size - 1, 0, -1):
        spaces = " " * (size - i)
        stars = "*" * (2 * i - 1)
        print(spaces + stars)

print("Triangle:")
draw_triangle(5)

print("\nRectangle:")
draw_rectangle(10, 4)

print("\nDiamond:")
draw_diamond(5)
